def sigmoid(g):

#可以封装成一个函数，命名为梯度下降/上升
def gradDescent(dataMatIn, classLabels):
    X = mat(dataMatIn)                                   
    y = mat(classLabels).transpose()              
    m,n = shape(X)                                       
    alpha = 0.001
    maxCycle = 500
    beta = ones((n,1))                                   
    for k in range(maxCycle):                     
        p = sigmoid(X*beta)                              
        error = (y - p)                                  
        beta = beta + alpha * X.transpose()* error  
    return beta

def gradDescentWithDraw(dataMatIn, classLabels):
    fig = plt.figure(figsize=(10,10))#设置画布
    ax = fig.add_subplot(311,ylabel='beta0')
    bx = fig.add_subplot(312,ylabel='beta1')
    cx = fig.add_subplot(313,ylabel='beta2')
    X = mat(dataMatIn)
    y = mat(classLabels).transpose()
    m,n = shape(X)
    alpha = 0.001
    maxCycle = 500
    beta = ones((n,1))              
    beta0 = []
    beta1 = []
    beta2 = []
    for k in range(maxCycle):
        h = sigmoid(X*beta)
        error = (y - h)
        beta += alpha * X.transpose() * error
        beta0.extend(beta[0])#记录beta0每一次迭代后的结果,下同
        beta1.extend(beta[1])
        beta2.extend(beta[2])
    ax.plot(range(maxCycle), beta0)
    bx.plot(range(maxCycle), beta1)
    cx.plot(range(maxCycle), beta2)
    plt.xlabel('iter_num')
    plt.show()
    return beta

3.1 实现随机梯度下降

In [21]:
def StocGradDescent(dataMatIn, classLabels, numIter=150):
    m,n = shape(dataMatIn)                                                      #m=100，n=3
    alpha = 0.001
    beta = ones(n)
    
    for j in range(numIter):                                                       #一共循环150*100次
        dataIndex = range(m)
        for i in range(m):
            randIndex = int(random.uniform(0,len(dataIndex)))#随机选中一条数据记录
            h = sigmoid(sum(dataMatIn[randIndex]*beta))       #Python中向量乘法不自动求和，所以要加sum
            error = classLabels[randIndex] - h
            beta = beta + alpha * error * dataMatIn[randIndex]#每次更新β，只使用一条数据，并用该误差更新所有权值
            del(dataIndex[randIndex])                                          #删除选中的数据，避免重复
            
    return beta
  • Python中向量、矩阵乘法的一些设定：
    • 向量乘法不自动求和: (1L,3L)*(1L,3L)=(1L,3L)，想要求和外面加sum()
    • 矩阵乘法自动求和: (2L,5L)*(5L,2L)=(2L,2L)
    • 矩阵乘法要么用np.dot(A,B)，要么用mat(A)*mat(B)

def StocGradDescentWithDraw(dataMatrix, classLabels, numIter=150):
    fig = plt.figure(figsize=(10,10))
    ax = fig.add_subplot(311,ylabel='beta0')
    bx = fig.add_subplot(312,ylabel='beta1')
    cx = fig.add_subplot(313,ylabel='beta2')
    m,n = shape(dataMatrix)     
    
    alpha = 0.001
    beta = ones(n)
    beta0 = array([])
    beta1 = array([])
    beta2 = array([])
    for j in range(numIter):
        dataIndex = range(m)
        for i in range(m):
            randIndex = int(random.uniform(0,len(dataIndex)))
            h = sigmoid(sum(dataMatrix[randIndex]*beta))
            error = classLabels[randIndex] - h
            beta = beta + alpha * error * dataMatrix[randIndex]
            del(dataIndex[randIndex])
            beta0 =append(beta0, beta[0])
            beta1 =append(beta1, beta[1])
            beta2 =append(beta2, beta[2])
    ax.plot(array(range(len(beta0))), beta0)
    bx.plot(array(range(len(beta1))), beta1)
    cx.plot(array(range(len(beta2))), beta2)
    plt.xlabel('iter_num')
    plt.show()
